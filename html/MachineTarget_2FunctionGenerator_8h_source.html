20 FunctionGenerator(llvm::LLVMContext &ctx, llvm::Module &module,
21 LLVMTypeGenerator &typeGen,
22 const std::string &name, const std::vector<std::string> &params,
23 FunctionType *type,
24 bool external = false)
25 : C(ctx), M(module), TC(typeGen), entryBBuilder(ctx), builder(ctx) {
26
27 auto funcType = llvm::dyn_cast<llvm::FunctionType>(type->createLLVMType(TC));
28 Func = llvm::Function::Create(funcType, llvm::Function::ExternalLinkage, name, module);
29
30 if (external) return;
31
32 auto entry = llvm::BasicBlock::Create(ctx, "entry", Func);
33 entryBBuilder.SetInsertPoint(entry);
34 auto body = llvm::BasicBlock::Create(ctx, "body", Func);
35// entryBBuilder.CreateBr(body);
36 builder.SetInsertPoint(body);
37 body->moveAfter(entry);
38
39 size_t idx = 0;
40 for (auto &arg: Func->args()) {
41 arg.setName(params[idx]);
42 createLocalVariable(std::string(arg.getName()), funcType->getParamType(idx), &arg);
43 ++idx;
44 }
45 }

47 void createLocalVariable(const std::string &name, llvm::Type *ty,
48 llvm::Value *initializer = nullptr) {
49 createEntryAlloca(ty, name);
50 if (initializer) storeToAlloca(name, initializer);
51 }
52
53 void createReturnValue(llvm::Value *value = nullptr) {
54 if (value) builder.CreateRet(value);
55 builder.CreateRetVoid();
56 }
57
58 llvm::Function *getFunction() { return Func; }

60 private:
61 llvm::AllocaInst *createEntryAlloca(llvm::Type *ty, const std::string &name) {
62 if (symbolTable.contains(name))
63 throw std::runtime_error{"Symbol already exists: " + name};
64 auto alloca = entryBBuilder.CreateAlloca(ty, nullptr, name);
65 symbolTable[name] = alloca;
66 return alloca;
67 }
68
69 llvm::LoadInst *loadFromAlloca(const std::string &name) {
70 if (!symbolTable.contains(name))
71 throw std::runtime_error{"Symbol doesn't exist: " + name};
72 auto alloca = symbolTable[name];
73 auto ty = alloca->getAllocatedType();
74 return builder.CreateLoad(ty, alloca, name);
75 }
76
77 void storeToAlloca(const std::string &name, llvm::Value *value) {
78 if (!symbolTable.contains(name))
79 throw std::runtime_error{"Symbol doesn't exist: " + name};
80 auto alloca = symbolTable[name];
81 builder.CreateStore(value, alloca);
82 }
